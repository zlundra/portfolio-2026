# Vitamin Deficiency & Disease — Surveillance-Style EDA (single copy/paste block)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import re

pd.set_option("display.max_columns", 100)
pd.set_option("display.width", 140)

DATA_PATH = "vitamin_deficiency_disease_dataset_20260123.csv"  # change if needed

# -----------------------------
# 1) Load
# -----------------------------
df_raw = pd.read_csv(DATA_PATH)
print("Shape:", df_raw.shape)
print(df_raw.head())

# -----------------------------
# 2) Standardize columns & QA
# -----------------------------
def normalize_col(c: str) -> str:
    c = str(c).replace("\u00a0", " ")
    c = re.sub(r"\s+", " ", c).strip().lower()
    c = c.replace(" ", "_")
    return c

df = df_raw.copy()
df.columns = [normalize_col(c) for c in df.columns]
print("\nColumns:", list(df.columns))

dup_n = int(df.duplicated().sum())
print("Duplicates:", dup_n)
df = df.drop_duplicates().copy()

def to_numeric_clean(s: pd.Series) -> pd.Series:
    return pd.to_numeric(s.astype(str).str.replace(r"[^\d\.\-]", "", regex=True), errors="coerce")

# Expected numeric columns for this dataset (only coerces if present)
numeric_like = [
    "age","bmi",
    "vitamin_a_percent_rda","vitamin_c_percent_rda","vitamin_d_percent_rda","vitamin_e_percent_rda",
    "vitamin_b12_percent_rda","folate_percent_rda","calcium_percent_rda","iron_percent_rda",
    "hemoglobin_g_dl","serum_vitamin_d_ng_ml","serum_vitamin_b12_pg_ml","serum_folate_ng_ml",
    "symptoms_count","has_multiple_deficiencies"
]
binary_cols = [c for c in df.columns if c.startswith("has_")]

for c in numeric_like:
    if c in df.columns:
        df[c] = to_numeric_clean(df[c])

# Convert binary has_* columns to bool when values are 0/1
for c in binary_cols:
    if c in df.columns:
        uniq = set(df[c].dropna().unique().tolist())
        if uniq.issubset({0, 1}):
            df[c] = df[c].astype(int).astype(bool)

print("\nInfo:")
print(df.info())

# -----------------------------
# 3) Missingness + quick dists
# -----------------------------
missing = (df.isna().mean() * 100).sort_values(ascending=False)
missing_df = missing.to_frame("missing_%")
print("\nTop missing columns:")
print(missing_df[missing_df["missing_%"] > 0].head(20))

top_missing = missing_df[missing_df["missing_%"] > 0].head(12)
if len(top_missing) == 0:
    print("No missing values detected.")
else:
    plt.figure(figsize=(10, 4))
    plt.bar(top_missing.index, top_missing["missing_%"])
    plt.title("Missing Values (%) — Top Columns")
    plt.xticks(rotation=45, ha="right")
    plt.ylabel("Missing (%)")
    plt.show()

for col in ["age", "bmi", "symptoms_count"]:
    if col in df.columns:
        plt.figure(figsize=(8, 4))
        plt.hist(df[col].dropna(), bins=35)
        plt.title(f"Distribution of {col}")
        plt.xlabel(col)
        plt.ylabel("Count")
        plt.show()

# -----------------------------
# 4) Feature engineering
# -----------------------------
# Age bands (adult-focused)
if "age" in df.columns:
    age_bins = [17, 25, 35, 45, 55, 65, 120]
    age_labels = ["18-25", "26-35", "36-45", "46-55", "56-65", "66+"]
    df["age_group"] = pd.cut(df["age"], bins=age_bins, labels=age_labels, include_lowest=True)
else:
    df["age_group"] = np.nan

# Nutrient %RDA columns
vit_cols = [c for c in df.columns if c.endswith("_percent_rda")]

# Below-RDA flags: %RDA < 100
for c in vit_cols:
    df[c + "_below_rda"] = df[c] < 100

below_cols = [c + "_below_rda" for c in vit_cols]
if len(below_cols) > 0:
    df["below_rda_count"] = df[below_cols].sum(axis=1)
    df["any_below_rda"] = df["below_rda_count"] > 0
    df["below_rda_severity"] = pd.cut(
        df["below_rda_count"],
        bins=[-0.1, 0.5, 2.5, 5.5, 100],
        labels=["None", "Mild(1-2)", "Moderate(3-5)", "Severe(6+)"]
    )
else:
    df["below_rda_count"] = np.nan
    df["any_below_rda"] = np.nan
    df["below_rda_severity"] = np.nan

print("\nEngineered preview:")
print(df[["age_group", "below_rda_count", "below_rda_severity"]].head())

# -----------------------------
# 5) Core visuals (surveillance-ready)
# -----------------------------
# 5.1 Top diagnoses
if "disease_diagnosis" in df.columns:
    diag = df["disease_diagnosis"].value_counts().head(10)
    plt.figure(figsize=(10, 4))
    plt.bar(diag.index, diag.values)
    plt.title("Top Diagnoses (count)")
    plt.xticks(rotation=45, ha="right")
    plt.ylabel("Count")
    plt.show()

# 5.2 Prevalence: %RDA < 100 by nutrient
if len(vit_cols) > 0:
    prev = {c.replace("_percent_rda", ""): float((df[c] < 100).mean() * 100) for c in vit_cols}
    prev_s = pd.Series(prev).sort_values(ascending=False)

    plt.figure(figsize=(10, 4))
    plt.bar(prev_s.index, prev_s.values)
    plt.title("Prevalence of %RDA < 100 by Nutrient")
    plt.xticks(rotation=45, ha="right")
    plt.ylabel("% of population")
    plt.show()

    print("\nTop nutrients by % below RDA:")
    print(prev_s.head(10).to_frame("pct_below_rda"))

# 5.3 Multiple deficiencies by age_group / latitude_region
if "has_multiple_deficiencies" in df.columns and df["has_multiple_deficiencies"].dtype == bool:
    if "age_group" in df.columns:
        by_age = df.groupby("age_group")["has_multiple_deficiencies"].mean() * 100
        plt.figure(figsize=(8, 4))
        plt.bar(by_age.index.astype(str), by_age.values)
        plt.title("Multiple Deficiencies (%) by Age Group")
        plt.ylabel("% with multiple deficiencies")
        plt.xticks(rotation=45, ha="right")
        plt.show()

    if "latitude_region" in df.columns:
        by_reg = df.groupby("latitude_region")["has_multiple_deficiencies"].mean() * 100
        plt.figure(figsize=(6, 4))
        plt.bar(by_reg.index.astype(str), by_reg.values)
        plt.title("Multiple Deficiencies (%) by Latitude Region")
        plt.ylabel("% with multiple deficiencies")
        plt.show()

# 5.4 Heatmap: multiple deficiencies rate by (age_group x latitude_region)
if "latitude_region" in df.columns and "has_multiple_deficiencies" in df.columns and df["has_multiple_deficiencies"].dtype == bool:
    pivot = df.pivot_table(
        index="age_group",
        columns="latitude_region",
        values="has_multiple_deficiencies",
        aggfunc="mean"
    ) * 100

    plt.figure(figsize=(7, 4))
    plt.imshow(pivot.values, aspect="auto")
    plt.xticks(range(pivot.shape[1]), pivot.columns.astype(str))
    plt.yticks(range(pivot.shape[0]), pivot.index.astype(str))
    plt.title("Multiple Deficiencies (%) — Age Group x Latitude Region")
    plt.colorbar(label="%")
    plt.show()

    print("\nPivot (% multiple deficiencies):")
    print(pivot)

# 5.5 Serum vs intake proxy: Vitamin D
if "vitamin_d_percent_rda" in df.columns and "serum_vitamin_d_ng_ml" in df.columns:
    d = df.dropna(subset=["vitamin_d_percent_rda", "serum_vitamin_d_ng_ml"]).copy()
    plt.figure(figsize=(7, 5))
    plt.scatter(d["vitamin_d_percent_rda"], d["serum_vitamin_d_ng_ml"], alpha=0.35)
    plt.title("Vitamin D %RDA vs Serum Vitamin D (ng/mL)")
    plt.xlabel("Vitamin D (%RDA)")
    plt.ylabel("Serum Vitamin D (ng/mL)")
    plt.show()

# 5.6 Serum Vitamin D by sun exposure
if "sun_exposure" in df.columns and "serum_vitamin_d_ng_ml" in df.columns:
    tmp = df.dropna(subset=["sun_exposure", "serum_vitamin_d_ng_ml"]).copy()
    labels = sorted(tmp["sun_exposure"].unique())
    groups = [tmp.loc[tmp["sun_exposure"] == g, "serum_vitamin_d_ng_ml"].values for g in labels]

    plt.figure(figsize=(8, 4))
    plt.boxplot(groups, labels=labels, showfliers=False)
    plt.title("Serum Vitamin D by Sun Exposure (outliers hidden)")
    plt.ylabel("Serum Vitamin D (ng/mL)")
    plt.xticks(rotation=45, ha="right")
    plt.show()

# 5.7 Symptom prevalence (has_* symptom columns)
symptom_cols = [
    "has_night_blindness","has_fatigue","has_bleeding_gums","has_bone_pain","has_muscle_weakness",
    "has_numbness_tingling","has_memory_problems","has_pale_skin"
]
symptom_cols = [c for c in symptom_cols if c in df.columns]

if len(symptom_cols) > 0:
    symp = df[symptom_cols].copy()
    for c in symptom_cols:
        if symp[c].dtype != bool:
            symp[c] = symp[c].astype(int).astype(bool)

    prev = symp.mean().sort_values(ascending=False) * 100
    plt.figure(figsize=(10, 4))
    plt.bar(prev.index, prev.values)
    plt.title("Symptom Prevalence (%)")
    plt.ylabel("% of population")
    plt.xticks(rotation=45, ha="right")
    plt.show()

    print("\nSymptom prevalence (%):")
    print(prev.to_frame("pct").head(10))

# -----------------------------
# 6) Correlations (numeric scan)
# -----------------------------
num_cols = [
    "age","bmi","symptoms_count","below_rda_count",
    "hemoglobin_g_dl","serum_vitamin_d_ng_ml","serum_vitamin_b12_pg_ml","serum_folate_ng_ml"
] + vit_cols
num_cols = [c for c in num_cols if c in df.columns]

if len(num_cols) >= 2:
    corr = df[num_cols].corr(numeric_only=True)

    plt.figure(figsize=(10, 6))
    plt.imshow(corr.values, aspect="auto")
    plt.xticks(range(len(num_cols)), num_cols, rotation=90)
    plt.yticks(range(len(num_cols)), num_cols)
    plt.title("Correlation Heatmap (numeric features)")
    plt.colorbar()
    plt.show()

    print("\nCorrelation (sample):")
    print(corr.iloc[:8, :8])

# -----------------------------
# 7) Surveillance KPIs
# -----------------------------
kpis = {
    "total_records": int(len(df)),
}

if "any_below_rda" in df.columns:
    kpis["pct_any_below_rda_(%RDA<100)"] = float(df["any_below_rda"].mean() * 100)

if "below_rda_count" in df.columns:
    kpis["mean_below_rda_count"] = float(df["below_rda_count"].mean())
    kpis["median_below_rda_count"] = float(df["below_rda_count"].median())

if "has_multiple_deficiencies" in df.columns and df["has_multiple_deficiencies"].dtype == bool:
    kpis["pct_multiple_deficiencies"] = float(df["has_multiple_deficiencies"].mean() * 100)

print("\n--- Surveillance KPIs ---")
for k, v in kpis.items():
    print(f"{k}: {v}")

if "disease_diagnosis" in df.columns:
    print("\nTop diagnoses:")
    print(df["disease_diagnosis"].value_counts().head(5))

print("\n✅ Data preparation + surveillance EDA complete. Ready for dashboard ingestion.")
