01_row_count_reconciliation.sql · SQL Server (source)
T-SQL
-- Row count reconciliation: source vs target per entity
SELECT
    'tbl_Registrant'   AS entity,
    COUNT(*)          AS source_count,
    42831             AS target_count,
    COUNT(*) - 42831 AS delta,
    CASE WHEN ABS(COUNT(*) - 42831) = 0   THEN 'PASS'
         WHEN ABS(COUNT(*) - 42831) <= 5  THEN 'WARN'
         ELSE 'FAIL'
    END AS status
FROM CRM_PROD_2019.dbo.tbl_Registrant
WHERE IsDeleted = 0

UNION ALL

SELECT
    'tbl_Application',
    COUNT(*),
    156318,
    COUNT(*) - 156318,
    CASE WHEN ABS(COUNT(*) - 156318) = 0   THEN 'PASS'
         WHEN ABS(COUNT(*) - 156318) <= 10 THEN 'WARN'
         ELSE 'FAIL'
    END
FROM CRM_PROD_2019.dbo.tbl_Application;
02_field_accuracy_spot_check.sql · SQL Server (source)
T-SQL
-- Field-level accuracy: compare key fields for matched records
-- Join on migrated external ID field (Legacy_ID__c in Salesforce)
SELECT
    src.RegistrantID,
    src.Surname        AS src_surname,
    tgt.LastName       AS tgt_surname,
    src.DateOfBirth    AS src_dob,
    tgt.Birthdate      AS tgt_dob,
    src.PINNumber      AS src_pin,
    tgt.PIN__c         AS tgt_pin,
    src.RegistrationStatus  AS src_status,
    tgt.Registration_Status__c AS tgt_status,
    CASE
        WHEN src.Surname <> tgt.LastName          THEN 'SURNAME_MISMATCH'
        WHEN src.DateOfBirth <> tgt.Birthdate      THEN 'DOB_MISMATCH'
        WHEN src.PINNumber <> tgt.PIN__c           THEN 'PIN_MISMATCH'
        ELSE 'MATCH'
    END AS field_check
FROM CRM_PROD_2019.dbo.tbl_Registrant src
INNER JOIN Salesforce_Snapshot.dbo.Contact tgt
    ON src.RegistrantID = tgt.Legacy_ID__c
WHERE
    src.Surname <> tgt.LastName
    OR src.DateOfBirth <> tgt.Birthdate
    OR src.PINNumber <> tgt.PIN__c
ORDER BY field_check, src.RegistrantID;
03_referential_integrity.sql · Cross-database check
T-SQL
-- Referential integrity: find orphaned application records
-- Applications in target that have no matching Contact
SELECT
    app.Id             AS sf_application_id,
    app.Legacy_ID__c   AS legacy_app_id,
    app.Contact__c     AS linked_contact_id,
    app.Application_Date__c,
    app.Status__c,
    'ORPHANED_APPLICATION' AS issue_type
FROM Salesforce_Snapshot.dbo.Application__c app
LEFT JOIN Salesforce_Snapshot.dbo.Contact con
    ON app.Contact__c = con.Id
WHERE con.Id IS NULL
    AND app.IsDeleted = 0
ORDER BY app.Application_Date__c DESC;

-- Result: 3 orphaned records found — blocking sign-off
04_null_rate_comparison.sql · Null rate audit
T-SQL
-- Null rate comparison: key fields source vs target
SELECT
    'Source' AS system,
    COUNT(*)                                        AS total,
    SUM(CASE WHEN PINNumber       IS NULL THEN 1 ELSE 0 END) AS null_pin,
    SUM(CASE WHEN DateOfBirth     IS NULL THEN 1 ELSE 0 END) AS null_dob,
    SUM(CASE WHEN EmailAddress    IS NULL THEN 1 ELSE 0 END) AS null_email,
    SUM(CASE WHEN AddressLine1    IS NULL THEN 1 ELSE 0 END) AS null_address
FROM CRM_PROD_2019.dbo.tbl_Registrant

UNION ALL

SELECT
    'Target',
    COUNT(*),
    SUM(CASE WHEN PIN__c          IS NULL THEN 1 ELSE 0 END),
    SUM(CASE WHEN Birthdate       IS NULL THEN 1 ELSE 0 END),
    SUM(CASE WHEN Email           IS NULL THEN 1 ELSE 0 END),
    SUM(CASE WHEN MailingStreet   IS NULL THEN 1 ELSE 0 END)
FROM Salesforce_Snapshot.dbo.Contact
WHERE Legacy_ID__c IS NOT NULL;
